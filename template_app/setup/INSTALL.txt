# install Miniconda3 https://conda.io/miniconda.html
# (I suggest not adding it to the PATH unless you have a reason -
#  the app works nicely without it)

# make sure conda is callble from terminal
# (I assume that you didn't change default Miniconda3 install dir)
#   on Windows:
cd /d %UserProfile%\Miniconda3\Scripts
set PATH=%PATH%;%cd%
#   on Unix (Linux/macOS):
cd ~/miniconda3/bin
export PATH="$PATH:$(pwd)"

# go to this dir
# run:
#   on Windows:
conda env create --file env_win.yaml
#   on Unix (Linux/macOS):
conda env create --file ./env_unix.yaml

# uninstall env:
conda env remove --name app
